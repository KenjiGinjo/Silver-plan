document.addEventListener("visibilitychange", function () {
  if (document.visibilityState === "visible") {
    // 页面变为可见，重新开始动画
    startAnimation();
  } else {
    // 页面不可见，停止动画
    stopAnimation();
  }
});

function startAnimation() {
  // 这里写启动动画的代码
  console.log("动画开始");
}

function stopAnimation() {
  // 这里写停止动画的代码
  console.log("动画停止");
}
